#define _CRT_SECURE_NO_WARNINGS 
#include <bits/stdc++.h>
using namespace std;
vector<pair<int, int>> num;
int x[200001];
int y[200001];
int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d %d", &x[i], &y[i]);
	for (int i = 0; i < n; i++)
		num.push_back(pair<int, int>(x[i], y[i]));
	sort(num.begin(), num.end());
	for (int i = 0; i < n; i++)
		printf("%d %d\n", num[i].first, num[i].second);
	system("pause");
	return 0;
}
